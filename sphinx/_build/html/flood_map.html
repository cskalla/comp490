from ipyleaflet import Map, GeoJSON, GeoData, basemaps, Marker, Heatmap,CircleMarker,LegendControl
import geopandas, pandas as pd, numpy as np

#enter map parameters
#center = (29.9511, -90.0615)
center = (29.992396263574907, -90.05232619149248)
zoom = 11
basemap = basemaps.CartoDB.Positron
#create map object
m10 = Map(center=center, zoom=zoom, basemap=basemap)

#load in income data from csv file
with open('data_High water mark collection Katrina .csv', 'r') as f:
    df2 = pd.read_csv(f, header=0)
df2 = df2.dropna(subset=['Survey Longitude'], axis = 0)
df2 = df2.dropna(subset=['Survey Latitude'], axis = 0)
df2 = df2.dropna(subset=['HWM Flood Elevation - NAVD88'], axis = 0)
df2.columns = [c.replace(' ', '_') for c in df2.columns]
df2 = df2[['Survey_Longitude', 'Survey_Latitude', 'HWM_Flood_Elevation_-_NAVD88']]
df2

import branca.colormap as cm
colormap = cm.linear.YlGnBu_03.to_step(data=df2['HWM_Flood_Elevation_-_NAVD88'], n=50)

long = df2['Survey_Longitude'].tolist()
lat = df2['Survey_Latitude'].tolist()
flood_level = df2['HWM_Flood_Elevation_-_NAVD88'].tolist()

data_frame = pd.DataFrame(
    {'Longitude': np.array(long),
      'Latitude': np.array(lat),
    'Flood_level': np.array(flood_level)
    })
gdf2 = geopandas.GeoDataFrame(
    data_frame, geometry=geopandas.points_from_xy(data_frame.Longitude, data_frame.Latitude))
